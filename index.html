  <script>

    /* ---------- KOD ---------- */

    let currentLayer = 'menu';

    const cursor = document.getElementById('cursor');

    const loader   = document.getElementById('loader');


    window.addEventListener('load', () => setTimeout(()=>loader.style.display='none', 1000));


    document.addEventListener('mousemove', e => {

      cursor.style.left = e.clientX + 'px';

      cursor.style.top  = e.clientY + 'px';

      if(Math.random() < .3) createParticle(e.clientX, e.clientY);

    });

    function createParticle(x,y){

      const p = document.createElement('div');

      p.classList.add('particle');

      p.style.left = x + 'px'; p.style.top = y + 'px';

      document.body.appendChild(p);

      setTimeout(()=>p.remove(),800);

    }

    function showLayer(id){

      document.querySelectorAll('.layer').forEach(l=>l.classList.remove('active'));

      document.getElementById(id).classList.add('active');

      currentLayer = id;

    }

    function startGame(){ showLayer('portfolio'); }

    document.addEventListener('keydown', e=>{

      switch(e.key){

        case 'ArrowRight': if(currentLayer==='menu') showLayer('portfolio'); break;

        case 'ArrowLeft':  showLayer('menu'); break;

        case 'ArrowUp':    if(currentLayer==='menu') showLayer('about'); break;

        case 'ArrowDown':  if(currentLayer==='menu') showLayer('contact'); break;

        case ' ': e.preventDefault(); if(currentLayer==='menu') startGame(); break;

      }

    });

  </script>
